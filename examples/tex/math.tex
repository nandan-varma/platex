\section{Inline and display math}
Euler's identity, $e^{i\pi} + 1 = 0$, relates five fundamental constants.
In display form:
\[
  e^{i\pi} + 1 = 0
\]

\section{Aligned equations}
\begin{align}
  (a + b)^2 &= a^2 + 2ab + b^2 \\
  (a - b)^2 &= a^2 - 2ab + b^2
\end{align}

\section{Sums, integrals, and limits}
\[
  \sum_{n=1}^{\infty} \frac{1}{n^2} = \frac{\pi^2}{6}
  \qquad
  \int_0^1 x^2 \, dx = \frac{1}{3}
  \qquad
  \lim_{x \to 0} \frac{\sin x}{x} = 1
\]

\section{Matrices and sets}

  \begin{pmatrix} x \\ y \end{pmatrix}
  =
  \begin{pmatrix} ax + by \\ cx + dy \end{pmatrix}
\]

  A \cup B, \quad A \cap B, \quad A \subseteq B, \quad \forall x \in \mathbb{R}, \; \exists y \in \mathbb{N}
